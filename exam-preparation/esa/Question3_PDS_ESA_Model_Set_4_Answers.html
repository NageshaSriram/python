Q3 (a) Stock Transaction Profit Margin James buys and sells Google LLC stock. We need to calculate purchase cost, commissions, selling price, and profit margin.

In[9]:
# Stock transaction calculation
units = 2000
purchase_price_per_unit = 40
sell_price_per_unit = 42.75
commission_rate = 0.03

# Purchase
purchase_cost = units * purchase_price_per_unit
purchase_commission = purchase_cost * commission_rate

# Sale
selling_price = units * sell_price_per_unit
selling_commission = selling_price * commission_rate

# Profit margin
profit = (selling_price - selling_commission) - (purchase_cost + purchase_commission)

print("Purchase Cost:", purchase_cost)
print("Purchase Commission:", purchase_commission)
print("Selling Price:", selling_price)
print("Selling Commission:", selling_commission)
print("Profit Margin:", profit)

def update_inventory(inventory, new_shipment):
    for product, qty in new_shipment.items():
        inventory[product] = inventory.get(product, 0) + qty
    return inventory

new_shipment = {"Laptop": 5, "Headphones": 20}
product_inventory = update_inventory(product_inventory, new_shipment)
print(product_inventory)

{'Laptop': 15, 'Phone': 25, 'Tablet': 15, 'Headphones': 20}

def apply_discount(inventory, discount_dict):
    for product, new_price in discount_dict.items():
        inventory[product] = f"Discounted Price: {new_price}"
    return inventory

discounts = {"Phone": 15000, "Tablet": 12000}

Q4 (a6) Categorize Athletes by Height & Weight

def categorize(row):
    if row['Height'] >= 180 and row['Weight'] >= 80:
        return 'Tall and Heavy'
    elif row['Height'] >= 180 and row['Weight'] < 80:
        return 'Tall and Light'
    elif row['Height'] < 180 and row['Weight'] >= 80:
        return 'Short and Heavy'
    else:
        return 'Short and Light'

merged['Category'] = merged.apply(categorize, axis=1)
print(merged[['Name','Height','Weight','Category']].head())

def player_profile(name):
    player_data = merged[merged['Name']==name]
    countries = player_data['region'].unique()
    sports = player_data['Sport'].unique()
